def url_by_url_type(url, url_type)
  case url_type

  when EXPORT
    Pathname(url).relative_path_from(Pathname('/')).to_s
  when FULL_URL
    URI::HTTP.build( SETTINGS['application']['default_url_options'].merge(path: url) ).to_s
  when SCORM
    File.join 'html', url
  when SCORM_HTML
    Pathname(url).relative_path_from(Pathname('/')).to_s
  else
    url
  end
end
