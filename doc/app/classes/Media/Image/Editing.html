def self.ratio_value(w, h, value)
  return value if h < RATIO_Y && w < RATIO_Y

  value.to_f *
    if ( w.to_f / h.to_f ) > ( RATIO_X / RATIO_Y )
      w / RATIO_X
    else
      h / RATIO_Y
    end
end
