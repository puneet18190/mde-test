def accept
  @ok = correct_integer? params[:id]
  if @ok
    @report = Report.find_by_id params[:id]
    @report.accept if @report
  end
end

def decline
  @ok = correct_integer? params[:id]
  if @ok

    @report.decline if @report
  end
