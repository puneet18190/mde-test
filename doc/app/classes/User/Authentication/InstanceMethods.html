def valid_password?(password)
  bcrypt = BCrypt::Password.new(encrypted_password)
  password = BCrypt::Engine.hash_secret("#{password}#{PEPPER}", bcrypt.salt)
  self.class.secure_compare(password, encrypted_password)
end
