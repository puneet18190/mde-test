def self.email?(email)
  index = 0
  at = false
  after_point = false
  count_after_point = 0
  after_at = false
  point_after_at = false
  email.each_byte do |i|
    if !Valid.char_for_email?(i)
      return false if index == 0
      if i == 64
        return false if at || after_point
        at = true
        after_at = true
        after_point = false
      elsif i == 46
        return false if after_at || after_point
        point_after_at = at
        after_point = true
        count_after_point = 0
      else
        return false
      end
    else
      after_at = false
      after_point = false
      count_after_point += 1
    end
    index += 1
  end
  return (point_after_at && count_after_point > 1)
end
