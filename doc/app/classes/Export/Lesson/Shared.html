def media_elements_files(options = {})
  exclude_versions = options[:exclude_versions] || []
  lesson.media_elements.map{ |r| r.media.paths.reject{ |k| exclude_versions.include?(k) }.values.map{ |v| Pathname(v) } }.flatten
end
